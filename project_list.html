후원 프로젝트 목록

import { storage, fmt } from "./js/app.js"; const wrap = document.querySelector("#list"); const sortSel = document.querySelector("#sort"); const onlyApproved = document.querySelector("#onlyApproved"); const render = () => { let items = storage.getAll(); if (onlyApproved.checked) items = items.filter(p => p.status === "approved"); if (sortSel.value === "latest") items.sort((a,b)=>b.createdAt - a.createdAt); if (sortSel.value === "progress") items.sort((a,b)=>{ const pa = a.targetAmount ? (a.raisedAmount||0)/a.targetAmount : 0; const pb = b.targetAmount ? (b.raisedAmount||0)/b.targetAmount : 0; return pb - pa; }); wrap.innerHTML = items.slice(0,6).map(p=>`

${p.title || ''}

${p.summary || ''}

목표 ${fmt(p.targetAmount)}원 구조자 ${fmt(p.selfPayAmount)}원
`).join(""); }; sortSel.addEventListener("change", render); onlyApproved.addEventListener("change", render); render();
