
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read: if true;
    }
    match /covers/{fileName} {
      allow write: if request.auth != null;
    }
    match /gallery/{fileName} {
      allow write: if request.auth != null;
    }
    match /receipts/{fileName} {
      allow write: if request.auth != null;
    }
  }
}
